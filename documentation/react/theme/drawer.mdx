# Tailwind CSS Drawer - Theme

<span id="overview" className="scroll-mt-48" />

Learn how to customize the theme and styles for drawer component, the theme object for drawer component has two main objects:

**A.** The <Code>defaultProps</Code> object for setting up the default value for props of drawer component.<br />
**B.** The <Code>styles</Code> object for customizing the theme and styles of drawer component.<br />

You can customize the theme and styles of drawer component by adding Tailwind CSS classes as key paired values for objects.

<br />
<br />

## Drawer Theme Object Type

<span id="theme-object-type" className="scroll-mt-48" />

```ts
interface DrawerStylesType {
  defaultProps: {
    size: number;
    overlay: boolean;
    placement: "top" | "right" | "bottom" | "left";
    overlayProps: React.ClassAttributes<HTMLDivElement> & React.HTMLAttributes<HTMLDivElement>;
    className: string;
    dismiss: {
      enabled: boolean;
      escapeKey: boolean;
      referencePress: boolean;
      referencePressEvent: 'pointerdown' | 'mousedown' | 'click';
      outsidePress: boolean | ((event: MouseEvent) => boolean);
      outsidePressEvent: 'pointerdown' | 'mousedown' | 'click';
      ancestorScroll: boolean;
      bubbles: boolean | {
          escapeKey: boolean;
          outsidePress: boolean;
      };
    };
    onClose: () => void;
    transition: object;
  };
  styles: {
    base: {
      container: object;
      overlay: object;
      drawer: object;
    };
  };
}
```

<br />
<br />

### For TypeScript Only

```tsx
import { DrawerStylesType } from "@material-tailwind/react";
```

## Drawer Theme Customization

<span id="theme-customization" className="scroll-mt-48" />

```ts
const theme = {
  drawer: {
    styles: {
      base: {
        drawer: {
          position: "fixed",
        },
        overlay: {
          position: "absolute",
        },
      },
    },
  },
};
```
