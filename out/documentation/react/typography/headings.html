React Headings

Different levels of headings for creating titles.

Heading 1

Material Tailwind

import React from "react";
import H1 from "@material-tailwind/react/Heading1";

export default function H1() {
  return (
    <H1 color="lightBlue">Material Tailwind</H1>
  );
}
